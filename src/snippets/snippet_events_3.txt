<link rel="import" href="../../bower_components/paper-button/paper-button.html">
<link rel="import" href="../../bower_components/paper-toast/paper-toast.html">
<link rel="import" href="../my-shared-styles.html">

<dom-module id="sample-events-3">
  <template>
    <style include="my-shared-styles"></style>
    <style>
      :host {
        display: block;
      }
    </style>
    <h2 class="red">[[title]]</h2>
    <p>Imperatively adding and removing an event listener.</p>
    <paper-button id="button" raised>Click me</paper-button>
    <paper-toast id="toast"></paper-toast>
  </template>
  <script>
    Polymer({
      is: 'sample-events-3',
      properties: {
        title: {
          type: String,
          value: 'sample-events-3',
        },
      },
      attached: function() {
        this.listen(this.$.button, 'tap', '_onTap');
      },
      detached: function() {
        this.unlisten(this.$.button, 'tap', '_onTap');
      },
      _onTap: function(event) {
        this.toast('You clicked on sample-events-3 button');
      },
      toast: function(text) {
        this.$.toast.text = text;
        this.$.toast.show();
      }
    });
  </script>
</dom-module>
